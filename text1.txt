local MapObjects = {Generators = {}, Pallets = {}, Windows = {}, Hooks = {}, Gates = {}}

local function RefreshMapObjects()
    if not isRunning then return end
    local temp = {Generators = {}, Pallets = {}, Windows = {}, Hooks = {}, Gates = {}}
    local processed = {}
    
    pcall(function()
        for _, obj in ipairs(Workspace:GetDescendants()) do
            if not obj or processed[obj] then continue end
            
            local name = string.lower(obj.Name or "")
            local fullName = string.lower(obj:GetFullName())

            if fullName:find("lobby") or fullName:find("menu") or fullName:find("spawn") or fullName:find("shop") then continue end
            if LocalPlayer.Character and obj:IsDescendantOf(LocalPlayer.Character) then continue end
            if obj:FindFirstChildOfClass("Humanoid") then continue end

            -- 1. DETEKSI WINDOWS / VAULTS (Fixed: 2 Target sesuai gambar)
            if name == "window" and obj.Parent and string.lower(obj.Parent.Name) == "vaults" then
                if not processed[obj] then
                    table.insert(temp.Windows, obj)
                    processed[obj] = true
                end
            elseif name == "vaulttrigger" or name == "vaultpoint" then
                local targetWindow = obj.Parent
                if targetWindow and not processed[targetWindow] then
                    table.insert(temp.Windows, targetWindow)
                    processed[targetWindow] = true
                end
                
            -- 2. DETEKSI ESCAPE GATES
            elseif name:find("gate") then
                if obj:FindFirstChild("LeftGate", true) or obj:FindFirstChild("RightGate", true) or obj:FindFirstChild("LeftGate-end", true) then
                    local targetGate = obj
                    if obj:IsA("BasePart") and obj.Parent and obj.Parent:IsA("Model") and obj.Parent ~= Workspace then targetGate = obj.Parent end
                    if not processed[targetGate] then table.insert(temp.Gates, targetGate); processed[targetGate] = true end
                end
                
            -- 3. DETEKSI GENERATOR
            elseif name:find("generator") or name:find("engine") or name:find("motor") or name == "gen" or name:find("^gen_") or name:find("_gen") then
                local targetGen = obj
                if obj:IsA("BasePart") and obj.Parent and obj.Parent:IsA("Model") and obj.Parent ~= Workspace then targetGen = obj.Parent end
                if not processed[targetGen] then table.insert(temp.Generators, targetGen); processed[targetGen] = true end
                
            -- 4. DETEKSI PALLETS
            elseif name:find("palletwrong") and fullName:find("map") then
                local targetPallet = obj
                if obj:IsA("BasePart") and obj.Parent and obj.Parent:IsA("Model") and obj.Parent ~= Workspace then targetPallet = obj.Parent end
                if not processed[targetPallet] then table.insert(temp.Pallets, targetPallet); processed[targetPallet] = true end
            end
        end
    end)
    MapObjects = temp
end

pcall(RefreshMapObjects)
task.spawn(function() while isRunning do task.wait(2.5); RefreshMapObjects() end end)

local function GetGenProgress(gen)
    local highest = 0
    pcall(function()
        local prompt = gen:FindFirstChildWhichIsA("ProximityPrompt", true)
        if prompt then
            local txt = (prompt.ObjectText or "") .. (prompt.ActionText or "")
            local num = txt:match("(%d+)%%")
            if num then highest = tonumber(num) end
        end
        if highest == 0 then
            for attr, val in pairs(gen:GetAttributes()) do
                local aName = string.lower(attr)
                if (aName:find("prog") or aName:find("percent")) and type(val) == "number" then
                    if val > highest and val <= 100 then highest = math.floor(val) end
                end
            end
        end
    end)
    return highest
end

-- ==================== SERVER HOP CORE SYSTEMS ====================
local isHopping = false
local function FetchAndHop(mode)
    if isHopping then return end
    isHopping = true
    SendNotification("Server Hop", "Sedang mencari server...", 3)
    
    task.spawn(function()
        local ApiUrl = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?limit=100"
        if mode == "Small" then ApiUrl = ApiUrl .. "&sortOrder=Asc" else ApiUrl = ApiUrl .. "&sortOrder=Desc" end

        local requestFunc = syn and syn.request or http_request or request
        if not requestFunc then SendNotification("Error", "Executor tidak support HTTP request.", 4); isHopping = false; return end
        
        local success, response = pcall(function() return requestFunc({Url = ApiUrl, Method = "GET"}) end)
        if success and response.StatusCode == 200 then
            local data = HttpService:JSONDecode(response.Body)
            if data and data.data then
                local candidates = {}
                for _, s in ipairs(data.data) do
                    if type(s) == "table" and s.id ~= game.JobId and s.playing and s.maxPlayers and s.playing < s.maxPlayers and s.playing > 0 then
                        table.insert(candidates, s)
                    end
                end
                
                if #candidates > 0 then
                    if mode == "Small" then table.sort(candidates, function(a, b) return a.playing < b.playing end) else
                        for i = #candidates, 2, -1 do local j = math.random(i); candidates[i], candidates[j] = candidates[j], candidates[i] end
                    end
                    SendNotification("Sukses", "Teleportasi ke server dengan " .. candidates[1].playing .. " player...", 4)
                    TeleportService:TeleportToPlaceInstance(game.PlaceId, candidates[1].id, LocalPlayer)
                else SendNotification("Gagal", "Tidak nemu server yang cocok saat ini.", 4) end
            end
        else SendNotification("API Error", "Gagal menghubungi Roblox. Coba lagi.", 4) end
        isHopping = false
    end)
end

-- ==================== ENGINE ESP Core ====================
local function DrawESP(obj, color, isEnabled, isPlayer, isGenerator, textNamePlayer)
    pcall(function()
        if not obj or not obj.Parent then return end
        local targetPart = obj:IsA("BasePart") and obj or obj.PrimaryPart or obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChildWhichIsA("BasePart", true)
        if not targetPart then targetPart = obj end

        local highlight = obj:FindFirstChild("KM_ESP") or targetPart:FindFirstChild("KM_ESP")
        local textLabelGui = targetPart:FindFirstChild("KM_TXT") or obj:FindFirstChild("KM_TXT")

        if isEnabled then
            if not highlight then highlight = Instance.new("Highlight"); highlight.Name = "KM_ESP"; highlight.FillTransparency = 0.5; highlight.Parent = obj:IsA("Model") and obj or targetPart end
            highlight.FillColor = color; highlight.OutlineColor = color; highlight.Enabled = true

            local showText = false
            local finalText = ""

            if isPlayer then
                if Config["Tampilkan Nama Player & Jarak"] then 
                    finalText = textNamePlayer
                    pcall(function() local dist = math.floor((LocalPlayer.Character.HumanoidRootPart.Position - targetPart.Position).Magnitude); finalText = finalText .. " [" .. dist .. "m]" end)
                    showText = true 
                end
            elseif isGenerator then
                local prog = GetGenProgress(obj)
                finalText = prog .. "%"
                showText = true
            end

            if showText then 
                if not textLabelGui then
                    textLabelGui = Instance.new("BillboardGui"); textLabelGui.Name = "KM_TXT"; textLabelGui.Size = UDim2.new(0, 200, 0, 50); textLabelGui.AlwaysOnTop = true; textLabelGui.Adornee = targetPart; textLabelGui.Parent = targetPart; textLabelGui.StudsOffset = Vector3.new(0, isPlayer and 3.5 or 2.5, 0)
                    local txt = Instance.new("TextLabel", textLabelGui); txt.Name = "Text"; txt.Size = UDim2.new(1,0,1,0); txt.BackgroundTransparency = 1; txt.TextStrokeTransparency = 0; txt.Font = Enum.Font.GothamBold; txt.TextSize = 14
                end
                local txtLabel = textLabelGui:FindFirstChild("Text")
                if txtLabel then txtLabel.TextColor3 = color; txtLabel.Text = finalText end 
                textLabelGui.Enabled = true
            else 
                if textLabelGui then textLabelGui.Enabled = false end
            end
        else
            if highlight then highlight:Destroy() end
            if textLabelGui then textLabelGui:Destroy() end
        end
    end)
end

-- ==================== MOBILE QTE FIX (EVENT-BASED & SANGAT RINGAN) ====================
local activeSkillChecks = {}
local function HandleSkillCheck(ui)
    if not isRunning then return end
    if not (Config["Auto Skill Check [INSTAN]"] or Config["Auto Skill Check [LEGIT]"]) then return end
    if not ui:IsA("GuiObject") then return end

    local uiName = string.lower(ui.Name)
    if uiName:find("skill") or uiName:find("check") or uiName:find("qte") or uiName:find("circle") or uiName:find("hitpoint") then
        if activeSkillChecks[ui] then return end
        activeSkillChecks[ui] = true

        task.spawn(function()
            if Config["Auto Skill Check [LEGIT]"] then 
                task.wait(math.random(35, 55) / 100) 
            else
                task.wait(0.02)
            end

            pcall(function()
                if firesignal then
                    firesignal(ui.MouseButton1Click)
                    firesignal(ui.MouseButton1Down)
                    firesignal(ui.TouchTap)
                end
                local absPos = ui.AbsolutePosition
                local absSize = ui.AbsoluteSize
                if absPos.X > 0 and absPos.Y > 0 then
                    VirtualInputManager:SendTouchTapEvent(absPos.X + (absSize.X/2), absPos.Y + (absSize.Y/2), true, game)
                end
            end)
            task.wait(1.5)
            activeSkillChecks[ui] = nil
        end)
    end
end

local playerGui = LocalPlayer:WaitForChild("PlayerGui")
playerGui.DescendantAdded:Connect(HandleSkillCheck)

for _, ui in ipairs(playerGui:GetDescendants()) do
    HandleSkillCheck(ui)
end

local currentGenPrompt = nil

task.spawn(function()
    while task.wait(0.05) do 
        if not isRunning then break end 
        
        -- ESP Renders
        for _, p in ipairs(Players:GetPlayers()) do
            pcall(function()
                if p ~= LocalPlayer and p.Character then
                    local isKiller = false
                    if p.Team and (string.lower(p.Team.Name):find("killer") or string.lower(p.Team.Name):find("murder")) then isKiller = true end
                    if isKiller then DrawESP(p.Character, Color3.fromRGB(255, 0, 0), Config["ESP Killer"], true, false, p.Name.." [KILLER]")
                    else DrawESP(p.Character, Color3.fromRGB(0, 255, 0), Config["ESP Survivor"], true, false, p.Name) end
                end
            end)
        end

        if MapObjects.Generators then for _, gen in ipairs(MapObjects.Generators) do pcall(function() DrawESP(gen, Color3.fromRGB(255, 255, 0), Config["ESP Generator"], false, true, "") end) end end
        if MapObjects.Pallets then for _, pal in ipairs(MapObjects.Pallets) do pcall(function() DrawESP(pal, Color3.fromRGB(255, 150, 0), Config["ESP Pallet"], false, false, "") end) end end
        if MapObjects.Windows then for _, win in ipairs(MapObjects.Windows) do pcall(function() DrawESP(win, Color3.fromRGB(0, 255, 255), Config["ESP Window"], false, false, "") end) end end
        if MapObjects.Gates then for _, gate in ipairs(MapObjects.Gates) do pcall(function() DrawESP(gate, Color3.fromRGB(255, 0, 255), Config["ESP Gate"], false, false, "") end) end end

        -- AUTO TP & GEN RUSH LOGIC
        pcall(function()
            if Config["Auto TP & Kerjakan Gen"] and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local killerPos = nil
                for _, p in ipairs(Players:GetPlayers()) do
                    if p ~= LocalPlayer and p.Character and p.Team and (string.lower(p.Team.Name):find("killer") or string.lower(p.Team.Name):find("murder")) then
                        if p.Character:FindFirstChild("HumanoidRootPart") then killerPos = p.Character.HumanoidRootPart.Position end
                    end
                end
                
                local myPos = LocalPlayer.Character.HumanoidRootPart.Position
                local isKillerNear = killerPos and (myPos - killerPos).Magnitude < 40
                local targetGen, minDistance = nil, math.huge

                if MapObjects.Generators then
                    for _, gen in ipairs(MapObjects.Generators) do
                        local tpPart = gen:IsA("BasePart") and gen or gen.PrimaryPart or gen:FindFirstChildWhichIsA("BasePart", true)
                        if tpPart then
                            local genProg = GetGenProgress(gen)
                            if genProg < 100 then
                                local genPos = tpPart.Position
                                local distToGen = (myPos - genPos).Magnitude
                                if isKillerNear then
                                    if (genPos - killerPos).Magnitude > 60 and distToGen < minDistance then minDistance = distToGen; targetGen = gen end
                                else if distToGen < minDistance then minDistance = distToGen; targetGen = gen end end
                            end
                        end
                    end
                end

                if targetGen then
                    local promptObj = nil
                    for _, v in ipairs(targetGen:GetDescendants()) do if v:IsA("ProximityPrompt") then promptObj = v; break end end
                    local tpPart = targetGen:IsA("BasePart") and targetGen or targetGen.PrimaryPart or targetGen:FindFirstChildWhichIsA("BasePart", true)
                    
                    if tpPart then
                        if minDistance > 5 then
                            LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(tpPart.Position + Vector3.new(3, 0, 0))
                            LocalPlayer.Character.HumanoidRootPart.Velocity = Vector3.new(0,0,0)
                        else
                            if promptObj then 
                                promptObj.RequiresLineOfSight = false
                                promptObj.MaxActivationDistance = 50
                                
                                local hum = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
                                if hum then for _, anim in ipairs(hum:GetPlayingAnimationTracks()) do anim:Stop() end end
                                LocalPlayer.Character.HumanoidRootPart.Anchored = false
                                
                                if currentGenPrompt ~= promptObj then
                                    if currentGenPrompt then pcall(function() currentGenPrompt:InputHoldEnd() end) end
                                    currentGenPrompt = promptObj
                                    promptObj:InputHoldBegin()
                                end
                            end
                        end
                    end
                else if currentGenPrompt then pcall(function() currentGenPrompt:InputHoldEnd() end); currentGenPrompt = nil end end
            else if currentGenPrompt then pcall(function() currentGenPrompt:InputHoldEnd() end); currentGenPrompt = nil end end
        end)

    end
end)
